from jakal_hwpx import HwpxDocument

doc = HwpxDocument.open("졸업논문양식.hwpx")
doc.set_metadata(
    title="고닉 특성 분석",
    creator="수능치는아랴양"
)
doc.replace_text("{{제목}}", "고닉 특성 분석")
doc.set_preview_text("졸업논문 초록 요약...")
doc.save("논문_완성.hwpx", validate=True)

검증 파이프라인 ● 정상 
# 1. XML 구조 검증
errors = doc.xml_validation_errors()
assert not errors, errors

# 2. 참조 무결성 검증
ref_errors = doc.reference_validation_errors()
assert not ref_errors, ref_errors

# 3. 전체 구조 검증
val_errors = doc.validation_errors()
assert not val_errors, val_errors

# 4. 저장 후 재열기 검증
reopen_errors = doc.save_reopen_validation_errors(
    "output.hwpx"
)
assert not reopen_errors, reopen_errors
